Qu 1. Write a program to create a simple calculator using switch?

    const Calculator = (A, B, C) => 
    {
	 switch(A){
	   case '+':  return (B+C);
	              break;
	              
	   case '-':  return (B-C);
	              break;
	              
	   case '*':  return (B*C);
	              break;
	              
	   case '/':  return (B/C);
	              break;
	              
	   default :  return ("Enter valid operator from (+,-,*,/)")
	              break;
	}
    
};
